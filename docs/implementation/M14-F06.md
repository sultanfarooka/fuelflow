### Phase 1 — TenantMigrationHostedService

- [x] Create `server/FuelFlow.Infrastructure/Services/TenantMigrationHostedService.cs`:
  ```csharp
  public class TenantMigrationHostedService : IHostedService
  {
      private readonly IServiceScopeFactory _scopeFactory;
      private readonly ILogger<TenantMigrationHostedService> _logger;
      private readonly IConfiguration _configuration;

      public TenantMigrationHostedService(
          IServiceScopeFactory scopeFactory,
          ILogger<TenantMigrationHostedService> logger,
          IConfiguration configuration)
      {
          _scopeFactory = scopeFactory;
          _logger = logger;
          _configuration = configuration;
      }

      public async Task StartAsync(CancellationToken ct)
      {
          using var scope = _scopeFactory.CreateScope();
          var controlPlane = scope.ServiceProvider.GetRequiredService<ControlPlaneDbContext>();

          var tenants = await controlPlane.Tenants
              .Where(t => t.Status == TenantStatus.Active)
              .AsNoTracking()
              .ToListAsync(ct);

          var baseConnStr = _configuration.GetConnectionString("DefaultConnection")
              ?? throw new InvalidOperationException("DefaultConnection not configured.");

          foreach (var tenant in tenants)
          {
              try
              {
                  var tenantConnStr = new NpgsqlConnectionStringBuilder(baseConnStr)
                  {
                      Database = tenant.DatabaseName
                  }.ToString();

                  var options = new DbContextOptionsBuilder<AppDbContext>()
                      .UseNpgsql(tenantConnStr,
                          npgsql => npgsql.MigrationsHistoryTable("__EFMigrationsHistory_AppDb"))
                      .Options;

                  await using var db = new AppDbContext(options);
                  await db.Database.MigrateAsync(ct);
                  _logger.LogInformation("Applied migrations to tenant {Id} ({Db})",
                      tenant.Id, tenant.DatabaseName);
              }
              catch (Exception ex)
              {
                  _logger.LogError(ex,
                      "Tenant migration failed for {Id} ({Db}); app will continue",
                      tenant.Id, tenant.DatabaseName);
              }
          }
      }

      public Task StopAsync(CancellationToken ct) => Task.CompletedTask;
  }
  ```

### Phase 2 — dev.ps1 -ResetAll

- [x] Extend `scripts/dev.ps1` with a `-ResetAll` switch:
  ```powershell
  param(
      [switch]$ResetAll
  )

  $ErrorActionPreference = "Stop"
  $root = Resolve-Path (Join-Path $PSScriptRoot "..")

  if ($ResetAll) {
      Write-Host "==> Resetting all databases..." -ForegroundColor Yellow

      # List all tenant_* databases and drop each one
      $tenantDbs = docker exec fuelflow-db psql -U fuelflow -d postgres -t -c `
          "SELECT datname FROM pg_database WHERE datname LIKE 'tenant_%'" |
          ForEach-Object { $_.Trim() } |
          Where-Object { $_ -ne "" }

      foreach ($db in $tenantDbs) {
          Write-Host "  Dropping tenant DB: $db" -ForegroundColor DarkYellow
          docker exec fuelflow-db psql -U fuelflow -d postgres -c `
              "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname = '$db' AND pid <> pg_backend_pid(); DROP DATABASE IF EXISTS ""$db"";" | Out-Null
      }

      # Drop and recreate the control plane DB
      Write-Host "  Dropping control plane DB: fuelflow_dev" -ForegroundColor DarkYellow
      docker exec fuelflow-db psql -U fuelflow -d postgres -c `
          "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname = 'fuelflow_dev' AND pid <> pg_backend_pid(); DROP DATABASE IF EXISTS fuelflow_dev; CREATE DATABASE fuelflow_dev;" | Out-Null

      # Rebuild control plane schema
      Write-Host "==> Rebuilding control plane schema..." -ForegroundColor Cyan
      Push-Location "$root\server"
      .\db-update.ps1 -Context ControlPlane
      Pop-Location

      Write-Host "==> Reset complete. Tenant DBs will be provisioned on first onboarding." -ForegroundColor Green
      exit 0
  }

  # Normal start (unchanged)
  Start-Process powershell -ArgumentList "-NoExit", "-Command", "cd '$root\fuel-flow-web'; npm run dev"
  Start-Process powershell -ArgumentList "-NoExit", "-Command", "cd '$root\server\FuelFlow.Api'; dotnet run"
  ```
- [x] **Acceptance:** Running `./scripts/dev.ps1 -ResetAll` in a shell with
      Docker running drops all tenant DBs and `fuelflow_dev`, rebuilds the
      control plane schema (all migrations applied), and exits without starting
      servers.
